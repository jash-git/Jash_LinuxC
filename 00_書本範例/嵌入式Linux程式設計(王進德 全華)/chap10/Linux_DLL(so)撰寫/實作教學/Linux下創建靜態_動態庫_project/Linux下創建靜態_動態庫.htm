{
        std::printf("in a( int a) %dn", i);
        return 0;
}

-----------b.cpp----------------

{
        std::cout<<" in int b(char *s): "<<s<<std::endl;
        return 0;

----------main.cpp------------- -
#include <cstdio>
using namespace std;
int a(int i);
int b(char *s);

{
        a(5);
        b("ok");
        printf("in mainn");
        return 0;
}

oB
1、g生成ao bo文件
# g++ -c a.cpp b.cpp

2、生成archive文件libtest.a c-建r-⑽募加入libtest.a
#ar rc libtest.a ao bo

3、指定oBlibtest.aM行g
# g++ -o main main.cpp libtest.a

\行Y果
# ./a.out
in a(int a) 5
in int b(char *s): ok
in main

B
1、g生成ao bo文件-fPIC生成.o文件r用相Φ刂F代a位置oP

2、生成Blibtest.so
# g++ -shared -o libtest.so ao bo

3、指定Blibtest.soM行g
# g++ main.cpp ./libtest.so
