int average(int count,...)
{
        int sum=0;
        va_list  args;              
        va_start(args,count);   // 飑lて ぃw把计C
        int i;
        for(i = 0 ; i < count ; ++i)
        {
                int arg = va_arg(args,int);   // 弄把计, CΩ弄岽N|欷U@, 堠蟀鸭篇韩O, ゲ斗パ{Α]pv郅vQ快kPq
                sum+=arg;
        }

}
